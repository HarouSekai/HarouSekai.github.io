6x6 自然数入力 => 3x3 平均出力

    条件
  • 入力する数値は空白で区切ること。
  • 出力は数値と空白の組み合わせとすること。
  • 空白が過不足なく出力されること。
  
    input = gets
    hw = input.split(" ")
    h = hw[0].to_i
    w = hw[1].to_i
    nums = []
    ht = h / 3
    wt = w / 3
    hi = 0
    while hi < h
      wi = 0
      while wi < w
        ss = gets.split(" ")
        c = 0
        k = []
        ss.each do |s|
          if c == 0
            k = []
          end
          k << s.to_i
          c += 1
          wi += 1
          if c == 3
            nums << k
            c = 0
          end
        end
      end
      hi += 1
    end
    
    ave_n = 0
    hp = 0
    while hp < ht
      wp = 0
      while wp < wt
        ave_n += 1
        wp += 1
      end
      hp += 1
    end
    aves = []
    ave_n.times do
      d = []
      aves << d
    end
    
    aves_i = 0
    r = 0
    nums.each do |num|
      num.each do |n|
        aves[aves_i] << n
      end
      aves_i += 1
    
      if aves_i % wt == 0
        aves_i -= wt
        r += 1
      end
      if r == 3
        aves_i += wt 
        r = 0
      end
    end
    
    s = 0
    q = ""
    aves.each do |ave|
      e = 0
      ave.each do |a|
        e += a
      end
      e = e / 9
      print e
      s += 1
      if s == wt
        puts ""
        s = 0
        q = ""
      else
        print " "
      end
    end
