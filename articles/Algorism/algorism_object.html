  def registration_student(students)
    # 生徒の名前と年齢を登録
    student = {name: "", age: 0, n: 0, m: 0, e: 0}
    puts '生徒名を入力してください'
    student[:name] = gets.chomp
    puts '生徒の年齢を入力してください'
    student[:age] = gets.chomp.to_i
  
    # 登録した生徒の国語、数学、英語の点数を登録
    puts "国語の得点は？"
    student[:n] = gets.chomp.to_i
    puts "数学の得点は？"
    student[:m] = gets.chomp.to_i
    puts "英語の得点は？"
    student[:e] = gets.chomp.to_i
  
    students << student
  end
  
  def show_student_name(students)
    # 登録された生徒の名前を番号を振って表示
    c = 0
    students.each do |student|
      puts "#{c}: #{student[:name]}"
      c += 1
    end
    puts '見たい生徒の番号を入力してください'
    student_number = gets.chomp.to_i
  
    # 選択された生徒の名前、年齢、国語、数学、英語の点数を表示
    puts "名前: #{students[student_number][:name]}"
    puts "年齢: #{students[student_number][:age]}"
    puts "国語: #{students[student_number][:n]}"
    puts "数学: #{students[student_number][:m]}"
    puts "英語: #{students[student_number][:e]}"
  end
  
  students = []
  
  while true
    puts '行いたい項目を選択してください'
    puts '[1]点数を登録する'
    puts '[2]点数を確認する'
    puts '[3]終了する'
    input = gets.to_i
    if input == 1
      registration_student(students)
    elsif input == 2
      show_student_name(students)
    elsif input == 3
      exit
    else
      puts '無効な値です'
    end
  end
