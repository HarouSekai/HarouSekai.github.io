if文 

  fun main() {
    val trafficLightColor = "Black"

    if (trafficLightColor == "Red") {
        println("Stop")
    } else if (trafficLightColor == "Yellow") {
        println("Slow")
    } else if (trafficLightColor == "Green") {
        println("Go")
    } else {
        println("Invalid traffic-light color")
    }

  }

when文 
  
    fun main() {
      val trafficLightColor = "Amber"
  
      when (trafficLightColor) {
          "Red" -> println("Stop")
          "Yellow", "Amber" -> println("Slow")
          "Green" -> println("Go")
          else -> println("Invalid traffic-light color")
      }
    }

if/else と when を式にする 
  
    fun main() {
      val trafficLightColor = "Black"
  
      val message =
        if (trafficLightColor == "Red") "Stop"
        else if (trafficLightColor == "Yellow") "Slow"
        else if (trafficLightColor == "Green") "Go"
        else "Invalid traffic-light color"
  
      println(message)
    }

    fun main() {
      val trafficLightColor = "Amber"
  
      val message = when(trafficLightColor) {
          "Red" -> "Stop"
          "Yellow", "Amber" -> "Proceed with caution."
          "Green" -> "Go"
          else -> "Invalid traffic-light color"
      }
      println(message)
    }

null

null値非許容変数
nullにできない。例：String型

null値許容変数
nullにできる。例：Int型

null値非許容変数をnull値許容変数にする方法
型の末尾に?を付ける。

  
    fun main() {
      var favoriteActor: String? = "Sandra Oh"
      println(favoriteActor)
      //-> Sandra Oh

      favoriteActor = null
      println(favoriteActor)
      //-> null
    }

「.」演算子

null値非許容変数のメソッドとプロパティにアクセスする。 
  
    fun main() {
      var favoriteActor: String = "Sandra Oh"
      println(favoriteActor.length)
      //-> 9
    }

    fun main() {
      var favoriteActor: String? = "Sandra Oh"
      println(favoriteActor.length)
      //-> エラー
    }

セーフコール演算子（?.）

null値許容変数のメソッドやプロパティにアクセスできる。 
  
    fun main() {
      var favoriteActor: String? = "Sandra Oh"
      println(favoriteActor?.length)
      //-> 9
    }

非nullアサーション演算子（!!）

変数の値にかかわらずnullでないことを強制する。nullの場合にエラーになるので出番が限られる。 
  
    fun main() {
      var favoriteActor: String? = "Sandra Oh"
      println(favoriteActor!!.length)
      //-> 9
    }

if/else構文との併用

  
    fun main() {
      val favoriteActor: String? = "Sandra Oh"
  
      val lengthOfName = if(favoriteActor != null) {
        favoriteActor.length
      } else {
        0
      }
  
      println("The number of characters in your favorite actor's name is $lengthOfName.")
    }

エルビス演算子（?:）

セーフコール演算子と合わせて使用できる。エルビス演算子を使用することで、セーフコール演算子がnullを返した時のデフォルト値を追加できる。 
  
    fun main() {
      val favoriteActor: String? = "Sandra Oh"
  
      val lengthOfName = favoriteActor?.length ?: 0
  
      println("The number of characters in your favorite actor's name is $lengthOfName.")
    }
