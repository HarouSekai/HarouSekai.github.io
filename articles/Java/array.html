配列の指定範囲から新しい配列を作成する

    
public static int[] copyOfRange(配列, 開始インデックス番号, 終了インデックス番号 - 1)
    
  
    
package sample;
import java.util.Arrays;

public class Sample01_01 {

	public static void main(String[] args) {
		int[] a = {1,2,4,8,16,32};
		int[] b = Arrays.copyOfRange(a, 3, 6);
		System.out.println(Arrays.toString(b));
          //[8, 16, 32]
	}
}

２つの配列を連結して新しい配列を作成

    
arraycopy(Object src, int srcPos, Object dest, int destPos, int length)
    
  
    src - 転送元配列
    srcPos - ソース配列の開始位置
    dest - 転送先配列
    destPos - 転送先データ内の開始位置
    length - コピーされつ配列要素の数
  
    
    package sample;
    import java.util.Arrays;
    
    public class Sample01_01 {
    
      public static void main(String[] args) {
        int[] src1 = {28,14,35,19};
        int[] src2 = {17, 40, 23};
        int[] dest = new int[src1.length + src2.length];
        
        System.arraycopy(src1, 0, dest, 0, src1.length);
        System.arraycopy(src2, 0, dest, src1.length, src2.length);
        
        System.out.println(Arrays.toString(src1));
        System.out.println(Arrays.toString(src2));
        System.out.println(Arrays.toString(dest));
      }
    }
    //[28, 14, 35, 19]
    //[17, 40, 23]
    //[28, 14, 35, 19, 17, 40, 23]

配列の要素を昇順にソートする

    
    public static void sort(int[] a)
    
  
    a - ソートされる配列
  
    
package sample;
import java.util.Arrays;

public class Sample01_01 {

	public static void main(String[] args) {
		int[] a = {12, 24, 18, 35, 21};
		
		System.out.println(Arrays.toString(a));
    //[12, 24, 18, 35, 21]
		
		Arrays.sort(a);
		System.out.println(Arrays.toString(a));
    //[12, 18, 21, 24, 35]
	}
}
