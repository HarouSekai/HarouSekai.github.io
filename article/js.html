const buildHTML = (XHR) => {
	const item = XHR.response.post;
	const html = `
		<div class="post">
			<div class="post-date">
				投稿日時：${item.created_at}
			</div>
			<div class="post-content">
				${item.content}
			</div>
		</div>`;
	return html;
};
